# Getters and Setter

  <?php
  class Book
  {
    public $title;
    public $author;
    public $pages;
    private $rating;
    function __construct($atitle, $aAuthor, $aPages, $aRating)
    {
      $this->title = $atitle;
      $this->author =$aAuthor;
      $this->pages = $aPages;
      $this->rating = $aRating;
    }

    function getInfo ()
    {
      echo "$this->title <br> $this->author <br> $this->pages <br> $this->rating <br>";
    }

    function getRating()
    {
      return $this->rating;
    }

    function setRating($newRating)
    {
      return $this->rating = $newRating;
    }
  }

  $book1 = new Book("Harrey Potter","JK Rowling",323, "PG-13");
  $book1->setRating("lkadhjsflskah");
  $book1->getRating();
  $book1->getInfo();
  ?>
